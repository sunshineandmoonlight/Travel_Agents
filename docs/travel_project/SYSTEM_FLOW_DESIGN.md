-- 餐厅知识库
CREATE TABLE IF NOT EXISTS restaurant_knowledge (
    id SERIAL PRIMARY KEY,
    name VARCHAR(200) NOT NULL,
    city VARCHAR(50) NOT NULL,
    district VARCHAR(100),
    address TEXT,
    phone VARCHAR(50),
    cuisine_type VARCHAR(100)[],
    avg_price INTEGER,
    rating DECIMAL(2,1),
    recommended_dishes TEXT[],
    business_hours VARCHAR(100),
    booking_required BOOLEAN DEFAULT FALSE,
    description TEXT,
    tags VARCHAR(50)[],
    created_at TIMESTAMP DEFAULT NOW()
);

-- 景点详细信息
CREATE TABLE IF NOT EXISTS attraction_detail (
    id SERIAL PRIMARY KEY,
    name VARCHAR(200) NOT NULL,
    city VARCHAR(50) NOT NULL,
    category VARCHAR(50),  -- 历史遗迹/自然风光/主题公园等
    description TEXT,      -- 详细介绍
    history TEXT,         -- 历史背景
    highlights TEXT[],    -- 必看景点
    photo_spots TEXT[],   -- 最佳拍照点
    tickets TEXT[],       -- 门票信息 {type, price, booking_method}
    opening_hours JSONB,  -- 开放时间 {season, time}
    best_season VARCHAR(50),
    visit_duration VARCHAR(50),
    tips TEXT[],          -- 游览贴士
    nearby_restaurants INTEGER[],  -- 附近餐厅ID
    nearby_attractions INTEGER[],
    created_at TIMESTAMP DEFAULT NOW()
);

    booking_channels TEXT[],
